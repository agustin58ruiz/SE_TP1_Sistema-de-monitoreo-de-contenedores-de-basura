#include "motor.h"
#include "arm_book_lib.h"

static void angleCallbackControl();
static int tickCounter = 1;





Motor::Motor(): Angulo()
{
    ControlAngulo.attach


}

static void angleCallbackControl(){

     tickCounter++;
     if (Angulo.Read() == ON){
         if(tickCounter > onTime){
             tickCounter = 1; 
             if(offTime) Angulo = OFF;
         }
     } else {
         if(tickCounter > offTime){
             tickCounter = 1;
             if( onTime ) Angulo = ON;
         }
     }



}
